func stressTest(a: i32, b: i32, c: i32, d: i32) -> i32 [
    
    a = b + c * d;
    b = (a - c) * (d + 4);
    d = a * b - c / d;

    a = b * c;
    b = a / d;
    c = (a + b - d);
    d = c + 1;
    a = b + c - d;

    a = -b + c - (-d);
    a = b - -c;
    
    b = !(a < c);
    c = a > b && d < a;
    d = a || d;

    a = (a + b) * (c - d) / (a + 1);
    b = (a * (b + (c * (d + 2))));

    return a + b + c + d;
]
